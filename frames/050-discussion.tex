\begin{frame}
    \frametitle{Discussion}
    \framesubtitle{Gaps and Limitations}

    \begin{itemize}
        \item<1->Existing datasets and challenges focus primarily on intra-camera tracking
              \vspace{5pt}
        \item<2->No challenge focuses solely on real-time inter-camera tracking
              \vspace{5pt}
        \item<3->Intra-camera tracking frameworks not optimized for inter-camera tracking
              \vspace{5pt}

    \end{itemize}
